    /** Cleanup if window is resized so that the different sizes work properly **/
    $(window).resize(function () {
        if (parseInt($(window).width()) > EmToPixels(40, false)) {
            $("html").removeClass("openRight");
            CloseSideRight();
        }
        AdjustCart();
    });
    var cartInitOffset;
    $(document).ready(function () {
        cartInitOffset = $("#divCart").offset().top - parseInt($("#divCart").css("margin-top"));
    });
    function showDetails(element) {
        if (!$(element).parent().parent().parent().find(".divItemAttributes").hasClass("hide")) {
            $(element).parent().parent().parent().find(".divItemAttributes").addClass("hide");
            $(element).find("img").attr("src", "/Images/Subway/arrow_down.png");
        } else {
            $(element).parent().parent().parent().find(".divItemAttributes").removeClass("hide");
            $(element).find("img").attr("src", "/Images/Subway/arrow_up.png");
        }
    }
    function SetCartPosition() {
        if ($(window).width() > EmToPixels(40, false)) {
            $("#divCart").css("right", "").css("left", $("#divWrapper").offset().left + $("#divWrapper").width() - $("#divCart").width() - window.pageXOffset - parseInt($("#divCart").css("margin-right")) + 'px').css("float", "none").css("margin-left", "0");
        }
    }
    function AdjustCart() {
        if ($("#divButton").css("display") == "none") {
            if ($("#divCart").height() < $("#divMain").outerHeight(false) - 50) {
                if ($(window).width() > EmToPixels(40, false)) {
                    //if sidebar is aligned to the bottom of the page and the top of cart is above where it natually should be.
                    if ($("#divCart").css("bottom") != "" && $("#divCart").offset().top < cartInitOffset) {
                        $("#divCart").css("position", "")
                        .css("right", "")
                        .css("float", "")
                        .css("left", "")
                        .css("bottom", "")
                        .css("display", "")
                        .css("right", "")
                        .css("margin-right", "")
                        .css("bottom", "")
                        .css("width", "")
                        .css("top", "");
                    }
                    //if sidebar is less than window height and the page position is after the cart start position
                    if ($("#divCart").outerHeight(true) < $(window).height() && window.pageYOffset >= cartInitOffset) {
                        $("#divCart").css("position", "fixed").css("top", "0").css("bottom", "");
                        SetCartPosition();
                    }
                    //if sidebar is pushing past the footer, position the cart just above the footer
                    if ($("#divCart").offset().top + $("#divCart").outerHeight(true) >= $("#divFooterContainer").offset().top) {
                        $("#divCart").css("position", "fixed").css("top", "").css("bottom", window.pageYOffset + $(window).height() - $("#divFooterContainer").offset().top + 'px');
                        SetCartPosition();
                    }
                        //if the bottom of the page is past the bottom of the cart, align the cart to the bottom of the browser.
                    else if (window.pageYOffset + $(window).height() > cartInitOffset + $("#divCart").outerHeight(true) && $(window).height() < $("#divCart").outerHeight(true)) {
                        $("#divCart").css("position", "fixed").css("bottom", "0").css("top", "");
                        SetCartPosition();
                    }
                }
            }
        }
        if (window.pageYOffset < $("#divHeaderContainer").height() + $(".divWrapperBody").height()) {
            $("#divCart").css("position", "")
            .css("right", "")
            .css("float", "")
            .css("left", "")
            .css("bottom", "")
            .css("display", "")
            .css("right", "")
            .css("margin-right", "")
            .css("bottom", "")
            .css("width", "")
            .css("top", "");
        }
    }

    function OrderSaved() {
        $('#BtnSaveOrder2').text('Order Saved').addClass('btnDisabled');
        $('span.message-context-change').text('Warning! When changing the menu context, items may be removed from your cart. Also, this order will no longer be listed as a Saved Order. Do you wish to continue?'); //Change the message on the popup change to express, informing user that saved order will be removed
        jsModalPopup.show({
            Class: 'profile_success',
            Icon: 'icon_success',
            H2: 'Saved',
            Message: 'Your order has been saved. Any changes to your cart will be saved automatically. To complete your order through the Call Center, please provide them with the following reference number: ' + '<b>0E2FB609-D6</b>.',
            OKFunc: 'jsModalPopup.close();'
        });
        return false;
    }

    function ShowAjaxErrorMessage() {
        jsModalPopup.show({
            H2: 'Error',
            Message: 'We were unable to save your order at this time.',
            OKFunc: 'jsModalPopup.close();'
        });
        return false;
    }

    function SaveOrder() {
        if ($('#BtnSaveOrder2').hasClass('btnDisabled')) {
            return false;
        }
        $.ajax({
            url: '../Services/SaveOrder.ashx',
            type: "POST"
        })
        .done(function (result) {
            if (result) {
                OrderSaved();
            } else {
                ShowAjaxErrorMessage();
            }
        })
        .fail(ShowAjaxErrorMessage);
        return false;
    }

    $(window).on("scroll", AdjustCart);
