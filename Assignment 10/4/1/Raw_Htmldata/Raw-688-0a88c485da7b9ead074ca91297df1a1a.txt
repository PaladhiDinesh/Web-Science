    var getParagraph = function($node, index){
      var maxCount=index+1, contentLineCount=0, breakCount=0, isNewLine=true;
      
      var walkDomSiblings = function(n){
        var storeNode = function(o){
          contentLineCount++;
          if(contentLineCount===maxCount){
            return o;
          }
        };
        var trackNewLine = function(){
          if(contentLineCount){
            isNewLine = true;    
          }
        };
        
        do {
          switch( n.nodeName.toLowerCase() ){
            case 'br' :
              breakCount++;
              if(breakCount>1){
                trackNewLine();
              }
              break;
            case 'p' :
              trackNewLine();
            default :
              if( isNewLine ){
                var content = $(n).text();
                if(content && $.trim(content)!==''){
                  isNewLine = false;
                  breakCount = 0;
                  var found = storeNode(n);
                  if(found){
                    return found;  
                  }
                }
              }
          }
          /*console.log('n: '+n.nodeName);*/
        } while(n = n.nextSibling)
